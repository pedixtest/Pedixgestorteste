🔥 Regras Firestore necessárias
Cole estas regras no Console Firebase → Firestore → Regras:
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ─────────────────────────────────────────
    // HELPERS
    // ─────────────────────────────────────────

    function isSuperAdmin() {
      return request.auth != null
        && request.auth.uid == "SeZDrWEKVxNFT5zEnVRqqezKuyi1";
    }

    // Lojista autenticado — uid bate com authUid salvo no tenant
    function isLojista(tenantId) {
      return request.auth != null
        && request.auth.uid ==
           get(/databases/$(database)/documents/tenants/$(tenantId)).data.authUid;
    }

    function ordersAreOpen(tenantId) {
      let main = get(/databases/$(database)/documents/tenants/$(tenantId)/appdata/main).data;
      return main.ordersOpen == true;
    }

    function isValidOrder(d) {
      let hasRequired = d.keys().hasAll(['client','phone','items','total','createdAt']);
      let validTotal  = d.total is number && d.total > 0 && d.total < 99999;
      let validPhone  = d.phone is string && d.phone.size() >= 8 && d.phone.size() <= 20;
      let validClient = d.client is string && d.client.size() >= 2 && d.client.size() <= 80;
      let validItems  = d.items is list && d.items.size() > 0 && d.items.size() <= 30;
      let validObs    = !('obs' in d) || (d.obs is string && d.obs.size() <= 500);
      let validAddr   = !('addr' in d) || (d.addr is string && d.addr.size() <= 200);
      let validStatus = !('status' in d) || d.status == "Novo";
      let noAuthField = !('_auth' in d);
      let allowedKeys = d.keys().hasOnly([
        'client','phone','items','total','createdAt',
        'obs','addr','bairro','delivery','subtotal','discount',
        'coupon','payment','status','time','date','id',
        'orderType','mesa','platform',
        'destLat','destLng','origemLat','origemLng','prazoRetirada'
      ]);
      return hasRequired && validTotal && validPhone && validClient
          && validItems && validObs && validAddr && validStatus
          && noAuthField && allowedKeys;
    }

    // ─────────────────────────────────────────
    // SAAS — só superAdmin
    // ─────────────────────────────────────────

    match /saas/{document=**} {
      allow read, write: if isSuperAdmin();
    }

    // ─────────────────────────────────────────
    // TENANTS
    // ─────────────────────────────────────────

    match /tenants/{tenantId} {

      // Raiz: leitura pública (app precisa ler storeName, blocked), escrita restrita
      allow read: if true;
      allow write: if isSuperAdmin() || isLojista(tenantId);

      // private/: nunca acessível pelo app — só superAdmin
      match /private/{doc} {
        allow read, write: if isSuperAdmin();
      }

      // appdata/main: leitura pública (cardápio, config), escrita restrita
      match /appdata/main {
        allow read: if true;
        allow write: if isSuperAdmin() || isLojista(tenantId);
      }

      // orders/: cliente cria sem auth, lojista e superAdmin lêem e atualizam
      match /orders/{orderId} {
        allow create: if ordersAreOpen(tenantId) && isValidOrder(request.resource.data);
        allow read, update: if isSuperAdmin() || isLojista(tenantId);
        allow delete: if isSuperAdmin() || isLojista(tenantId);
      }

      // Resto do tenant (categorias, produtos, config extra)
      match /{subcollection=**} {
        allow read: if isSuperAdmin() || isLojista(tenantId);
        allow write: if isSuperAdmin() || isLojista(tenantId);
      }
    }

    // ─────────────────────────────────────────
    // CLIENTES — só superAdmin
    // ─────────────────────────────────────────

    match /clientes/{tenantId}/{document=**} {
      allow read, write: if isSuperAdmin();
    }

    // ─────────────────────────────────────────
    // MOTOBOYS — privado, sem leitura pública
    // ─────────────────────────────────────────

    match /motoboys/{motoboyId} {
      allow read: if isSuperAdmin()
        || (request.auth == null && resource.data.keys().hasOnly(['status','available','updatedAt']));
      allow write: if isSuperAdmin();
      // Auto-cadastro: apenas criação sem PIN (status pending)
      allow create: if request.resource.data.status == "pending"
        && request.resource.data.keys().hasAll(['nome','telefone','status','createdAt'])
        && !('_pin' in request.resource.data);
      // Motoboy atualiza próprio doc (localização, status) com PIN
      allow update: if request.resource.data._auth == resource.data._pin
        && resource.data._pin != ""
        && resource.data.status != "blocked";
    }
  }
}
